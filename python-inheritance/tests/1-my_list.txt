# Test cases for MyList class

# Test 1: Normal list of integers
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

# Test 2: Empty list
>>> empty_list = MyList()
>>> print(empty_list)
[]
>>> empty_list.print_sorted()
[]
>>> print(empty_list)
[]

# Test 3: List with one element
>>> single_element_list = MyList([42])
>>> print(single_element_list)
[42]
>>> single_element_list.print_sorted()
[42]
>>> print(single_element_list)
[42]

# Test 4: List with duplicate elements
>>> duplicate_list = MyList([3, 1, 2, 1, 4])
>>> print(duplicate_list)
[3, 1, 2, 1, 4]
>>> duplicate_list.print_sorted()
[1, 1, 2, 3, 4]
>>> print(duplicate_list)
[3, 1, 2, 1, 4]
